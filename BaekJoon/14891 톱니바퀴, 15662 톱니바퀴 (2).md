from collections import deque

t = int(input())
d = list(
    deque(list(map(int, input())))
    for _ in range(t)
)

k = int(input())
order = [
    list(map(int, input().split()))
    for _ in range(k)
]

for a, b in order:
    o, l, r = b, d[a - 1][6], d[a - 1][2]
    d[a - 1].rotate(b)

    for i in range(a - 2, -1, -1):
        if d[i][2] != l:
            l = d[i][6]
            b *= - 1
            d[i].rotate(b)
        else:
            break

    b = o
    for i in range(a, t):
        if r != d[i][6]:
            r = d[i][2]
            b *= -1
            d[i].rotate(b)
        else:
            break

answer = 0
for i in range(t):
    if d[i][0] == 1:
        answer += 1

print(answer)
